function ReviewCard({ review, isEven }: { review: Review; isEven: boolean }) {
  // 赤文字を適用する処理
  const highlightText = (text: string) => {
    return text.split(/(\【[^\】]+\】)/g).map((part, index) => {
      if (part.startsWith('【') && part.endsWith('】')) {
        return (
          <span key={index} style={{ color: '#ff3131', fontWeight: 'bold' }}>
            {part.replace(/【|】/g, '')}
          </span>
        );
      }
      return part;
    });
  };

  return (
    <div style={{
      display: 'flex',
      flexDirection: isEven ? 'row' : 'row-reverse',
      gap: '20px',
      backgroundColor: 'white',
      padding: '30px',

## 📞 トラブルシューティング

### エラー1: 赤文字が表示されない
**原因**: 【】が正しく認識されていない
**解決**: highlightText関数を確認、正規表現を修正

### エラー2: レイアウトが崩れる
**原因**: スタイルの競合
**解決**: インラインスタイルを優先、!important を使用

**原因**: 削除したファイルのインポートが残っている
**解決**: page.tsx と他のファイルのインポート文を確認

### エラー4: CTAボタンが動作しない
**原因**: onClick が正しく設定されていない
**解決**: CTAButton.tsx の onClick プロパティを確認
